### x만큼 간격이 있는 n개의 숫자 
```
function solution(x, n) {
    let answer = []
    for(let i = 1; i <= n; i++){
        answer.push(x * i)
    }
    return answer;
}
```

### 문자열 내 p와 y의 개수
```
function solution(s){
    let answer;
    const arr = [...s];
    let p = 0;
    let y = 0;
    for(let i = 0; i < arr.length; i++){
       arr[i] === "p" || arr[i] === "P" ? p +=1 :
       arr[i] === "y" || arr[i] === "Y" ? y +=1 : null
    }
    p === y ? answer = true : answer = false
    return answer;
}
```
